This file contains test cases for a class that inherits from the list 
class.
=====================================================================

>>> MyList = __import__("1-my_list").MyList
>>> myList = MyList()
>>> myList.append(2)
>>> myList.append(7)
>>> myList.append(0)
>>> myList.append(45)
>>> myList.append(8956)
>>> myList.append(-56)
>>> print(myList)
[2, 7, 0, 45, 8956, -56]

>>> myList.print_sorted()
[-56, 0, 2, 7, 45, 8956]

>>> myList.clear()
>>> print(myList)
[]

>>> myList.append(-56)
>>> myList.append(-55)
>>> myList.append(-1)
>>> myList.append(0)

[-56, -55, -2, -1, 0]

>>> myList.append(-1)
